% 四則演算 文法

式	: 項 '+' 式
	| 項 '-' 式
	| 項

項	: 因子 '*' 項
	| 因子 '/' 項
	| 因子

因子	: 数字
	| '(' 式 ')'



G2 {Vn, Vt, P, S}
Vn {expr, term, factor number}
Vt {+, -, *, /, (, ), digit}

expr	: term '+' expr
	| term '-' expr
	| term

term 	: factor '*' term
	| factor '/' term
	| factor

factor 	: number
	| '(' expr ')'

number  : '-' number
	| digit number
	| digit
